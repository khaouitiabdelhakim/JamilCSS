  { utility: "2xl:", css: "min-width: 1536px" },
]} />

</div>

<div className="j-grid j-grid-cols-1 md:j-grid-cols-2 lg:j-grid-cols-3 j-gap-6">
  Responsive grid
</div>`} />

Override defaults in `jamil.config.js` with `theme.breakpoints`.
